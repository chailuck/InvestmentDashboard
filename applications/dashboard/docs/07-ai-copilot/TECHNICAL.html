2.2 SSE event format

Each Server-Sent Event is a JSON-encoded string on a single data: line. The client parses the JSON after stripping the "data: " prefix.

data: {"type": "text",       "content": "Hello "}
data: {"type": "tool_start", "name": "get_portfolio_positions"}
data: {"type": "tool_end",   "name": "get_portfolio_positions"}
data: {"type": "error",      "content": "Error message string"}
data: {"type": "done",       "session_id": "uuid-string"}

2.3 Streaming generator

The generator runs a tool-use loop, calling the Anthropic API up to 6 times until the model stops requesting tools. Each Anthropic call is wrapped with asyncio.to_thread() because the SDK is synchronous.

async def _stream_response(message: str, session_id: str) -> AsyncIterator[str]:
    client = anthropic.Anthropic(api_key=settings.anthropic_api_key)
    messages = _SESSIONS.setdefault(session_id, [])
    messages.append({"role": "user", "content": message})

    for _iteration in range(6):    # max 6 tool-use iterations per message
        response = await asyncio.to_thread(
            client.messages.create,
            model=settings.ai_default_model,
            max_tokens=4096,
            system=_system_prompt(),
            messages=messages,
            tools=TOOLS,
        )

        # Append assistant response to session history
        messages.append({"role": "assistant", "content": response.content})

        # Stream text blocks word-by-word for a real-time effect
        for block in response.content:
            if block.type == "text":
                for word in block.text.split(" "):
                    yield f"data: {json.dumps({'type': 'text', 'content': word + ' '})}\n\n"
                    await asyncio.sleep(0)    # yield to event loop between words

        # If model stopped for a reason other than tool_use, we're done
        if response.stop_reason != "tool_use":
            break

        # Execute all tool calls in this response
        tool_results = []
        for block in response.content:
            if block.type != "tool_use":
                continue
            yield f"data: {json.dumps({'type': 'tool_start', 'name': block.name})}\n\n"
            result = await asyncio.to_thread(_execute_tool, block.name, block.input)
            yield f"data: {json.dumps({'type': 'tool_end', 'name': block.name})}\n\n"
            tool_results.append({
                "type": "tool_result",
                "tool_use_id": block.id,
                "content": str(result),
            })
        messages.append({"role": "user", "content": tool_results})

    # Trim session to last 20 messages to avoid unbounded memory growth
    _SESSIONS[session_id] = messages[-20:]
    yield f"data: {json.dumps({'type': 'done', 'session_id': session_id})}\n\n"

4. System prompt

The system prompt is built dynamically from files in /app/investment_agent/ on every API call. This means the prompt always reflects the current state of the knowledge base without requiring a restart.

def _system_prompt() -> str:
    parts = []
    base_dir = Path("/app/investment_agent")

    # 1. CLAUDE.md — main persona (if it exists)
    claude_md = base_dir / "CLAUDE.md"
    if claude_md.exists():
        parts.append(claude_md.read_text())
    else:
        parts.append("You are an expert investment advisor with deep knowledge of Thai SET stocks...")

    # 2. Personal trading style knowledge
    style_doc = base_dir / "knowledge" / "personal-trading-style.md"
    if style_doc.exists():
        parts.append(f"## Personal Trading Style\n{style_doc.read_text()}")

    # 3. Dashboard tool usage instructions
    parts.append("""
## Available Tools
Use the provided tools to fetch live data before answering data-dependent questions.
Always call get_portfolio_positions before answering questions about current holdings.
    """)

    return "\n\n".join(parts)

5. Frontend — streaming consumer

The AI Copilot uses the native Fetch API with a ReadableStream reader rather than Axios. Axios does not support streaming response bodies — it buffers the entire response before resolving the promise.

const sendMessage = async (message: string) => {
  const token = useAuthStore.getState().accessToken
  const response = await fetch('/api/proxy/api/v1/ai/copilot/stream', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Authorization': `Bearer ${token}`,
    },
    body: JSON.stringify({ message, session_id: sessionId }),
  })

  if (!response.ok) {
    throw new Error(`HTTP ${response.status}`)
  }

  const reader = response.body!.getReader()
  const decoder = new TextDecoder()
  let currentMessage = ''

  while (true) {
    const { done, value } = await reader.read()
    if (done) break

    const text = decoder.decode(value, { stream: true })

    // Parse SSE lines — each line is "data: {...}"
    for (const line of text.split('\n')) {
      if (!line.startsWith('data: ')) continue
      try {
        const event = JSON.parse(line.slice(6))
        switch (event.type) {
          case 'text':
            currentMessage += event.content
            setCurrentAssistantMessage(currentMessage)
            break
          case 'tool_start':
            setActiveTool(event.name)
            break
          case 'tool_end':
            setActiveTool(null)
            break
          case 'done':
            setSessionId(event.session_id)
            finalizeMessage(currentMessage)
            currentMessage = ''
            break
          case 'error':
            showError(event.content)
            break
        }
      } catch {
        // Ignore malformed SSE lines
      }
    }
  }
}

6. Session storage

Sessions are stored in a module-level dictionary in ai.py. This is a deliberate simplification — the AI Copilot is a personal tool with a single primary user, and SSE connections are sticky to one worker.

_SESSIONS: dict[str, list] = {}   # session_id → last 20 messages
LimitationImpactMitigation
Not shared across workersConnection to a different worker loses historySSE is sticky per connection; low impact in practice
Lost on restartAll conversation history clearedUser can start a new session manually; history is bounded to last 20 messages anyway
Max 20 messagesVery long conversations lose early contextImportant context is available via tools (portfolio data, knowledge docs)
