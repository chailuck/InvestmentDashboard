2. Allocation chart

File: AllocationChartWidget.tsx

The allocation chart fetches all active positions, computes the market value per symbol (summing multiple positions in the same stock), then builds an ECharts pie series with the donut style.

const { data } = useQuery({
  queryKey: ['portfolio-positions', 'active'],
  queryFn: () => portfolioTrackerService.getPositions({ status: 'active' }),
  refetchInterval: 60_000,
  staleTime: 60_000,
})

// Aggregate value per symbol (handles multiple positions in same stock)
const slices = positions.reduce((acc, pos) => {
  const value = pos.currentPrice * pos.positionSize
  acc[pos.symbol] = (acc[pos.symbol] ?? 0) + value
  return acc
}, {} as Record<string, number>)

// Build ECharts option
const option: EChartsOption = {
  backgroundColor: '#0d1117',
  series: [{
    type: 'pie',
    radius: ['40%', '70%'],    // donut style
    label: { color: '#c9d1d9' },
    data: Object.entries(slices).map(([name, value]) => ({ name, value })),
    emphasis: { itemStyle: { shadowBlur: 10 } }
  }]
}

3. Market indices

Two separate backend endpoints handle Thai and global indices. Both call Yahoo Finance's v7/finance/quote endpoint directly (not via the yfinance library) for lower latency.

# backend/app/services/portfolio_excel.py

SET_TICKERS    = ["^SET.BK", "^SET50.BK", "^SET100.BK", "^MAI.BK", "^sSET.BK"]
GLOBAL_TICKERS = ["^GSPC", "^IXIC", "^DJI", "BTC-USD", "GC=F"]

def _yahoo_quote_direct(tickers: list[str]) -> dict:
    url = "https://query1.finance.yahoo.com/v7/finance/quote"
    params = {
        "symbols": ",".join(tickers),
        "fields": "regularMarketPrice,regularMarketChange,regularMarketChangePercent,shortName"
    }
    resp = requests.get(url, headers={"User-Agent": "Mozilla/5.0"}, params=params, timeout=10)
    results = resp.json()["quoteResponse"]["result"]
    return {r["symbol"]: r for r in results}

def fetch_set_indices() -> list[dict]:
    quotes = _yahoo_quote_direct(SET_TICKERS)
    return [
        {
            "name": ticker.replace("^", "").replace(".BK", ""),
            "value": quotes.get(ticker, {}).get("regularMarketPrice"),
            "change": quotes.get(ticker, {}).get("regularMarketChange"),
            "changePct": quotes.get(ticker, {}).get("regularMarketChangePercent"),
        }
        for ticker in SET_TICKERS
    ]

The component manages its own range state internally. The zoomStartForRange helper computes the ECharts dataZoom.start percentage so the visible window covers the last N calendar days regardless of the total candle count in the dataset.

function zoomStartForRange(dates: string[], days: number): number {
  if (!dates.length || days >= 912) return 0
  const cutoff = new Date()
  cutoff.setDate(cutoff.getDate() - days)
  const cutoffStr = cutoff.toISOString().slice(0, 10)
  const idx = dates.findIndex(d => d.slice(0, 10) >= cutoffStr)
  if (idx <= 0) return 0
  return (idx / dates.length) * 100
}

5. PortfolioSummaryWidget

File: PortfolioSummaryWidget.tsx

Renders four animated metric cards. THB values are formatted with the fmtTHB helper which compacts large numbers into K/M notation:

function fmtTHB(n: number): string {
  const sign = n >= 0 ? '+' : ''
  if (Math.abs(n) >= 1_000_000) return `${sign}${(n / 1_000_000).toFixed(2)}M ฿`
  if (Math.abs(n) >= 1_000)     return `${sign}${(n / 1_000).toFixed(1)}K ฿`
  return `${sign}${n.toLocaleString('th-TH', { maximumFractionDigits: 0 })} ฿`
}

Each card uses Framer Motion for entry and value-change animations:

// Card entry animation
<motion.div
  initial={{ opacity: 0, y: 8 }}
  animate={{ opacity: 1, y: 0 }}
  transition={{ duration: 0.3, delay: index * 0.05 }}
>

// Value change animation (AnimatePresence)
<AnimatePresence mode="wait">
  <motion.span
    key={value}    // key change triggers exit/enter when value changes
    initial={{ opacity: 0, y: 6 }}
    animate={{ opacity: 1, y: 0 }}
    exit={{ opacity: 0, y: -6 }}
  >
    {value}
  </motion.span>
</AnimatePresence>

6. Dashboard layout

File: frontend/src/app/(dashboard)/layout.tsx

The shared layout for all dashboard pages uses a flex row with a fixed-width sidebar and a scrollable main content area:

export default function DashboardLayout({ children }: { children: React.ReactNode }) {
  const { user } = useAuthStore()

  // Client-side guard — redirect to /login if not authenticated
  useEffect(() => {
    if (!user) router.replace('/login')
  }, [user])

  return (
    <div className="flex h-screen bg-[#0d1117]">
      <Sidebar />
      <main className="flex-1 overflow-y-auto p-6">
        {children}
      </main>
    </div>
  )
}

The actual widget layout on the dashboard page uses Tailwind's CSS grid. The two-column layout with different widget sizes is achieved with grid-cols-2 and col-span-1 / col-span-2 classes.
