Settings & Admin — Technical Design

1. App configuration

Storage

Configuration is persisted to /app/uploads/app_config.json inside the backend container. This path is in the uploads_data named Docker volume, which is persistent across container restarts.

{
  "excel_source_path": "/app/investment_data/Investment tracking.xlsx",
  "excel_working_path": "/app/uploads/investment_tracking.xlsx"
}

File: backend/app/services/app_config_service.py

CONFIG_PATH = Path("/app/uploads/app_config.json")

def get_app_config() -> dict:
    if CONFIG_PATH.exists():
        return json.loads(CONFIG_PATH.read_text())
    return {}   # fallback: use environment variable defaults

def save_app_config(data: dict) -> None:
    CONFIG_PATH.parent.mkdir(parents=True, exist_ok=True)
    CONFIG_PATH.write_text(json.dumps(data, indent=2))

Endpoints

File: backend/app/api/v1/endpoints/app_config.py

@router.get("/app-config")
async def get_config(_: UserId) -> dict:
    cfg = get_app_config()
    # Merges stored config with environment variable defaults
    return {
        "excel_source_path": cfg.get("excel_source_path", settings.investment_excel_source_path),
        "excel_working_path": cfg.get("excel_working_path", settings.investment_excel_path),
    }

@router.put("/app-config")
async def save_config(body: AppConfigUpdate, _: UserId) -> dict:
    cfg = get_app_config()
    if body.excel_source_path is not None:
        cfg["excel_source_path"] = body.excel_source_path
    if body.excel_working_path is not None:
        cfg["excel_working_path"] = body.excel_working_path
    save_app_config(cfg)
    return {"status": "ok"}

@router.post("/app-config/test-path")
async def test_path(body: TestPathRequest, _: UserId) -> dict:
    p = Path(body.path)
    return {
        "exists": p.exists(),
        "readable": p.is_file() and os.access(p, os.R_OK),
        "size_kb": round(p.stat().st_size / 1024, 1) if p.exists() else None,
    }
App configuration is accessible to all authenticated users, not just admins. This was a deliberate change from an earlier version where it was admin-gated, to allow any user to configure their own Excel source path if they have multiple data sources.

2. User management

File: backend/app/api/v1/endpoints/users.py

All user management endpoints are protected by the require_admin FastAPI dependency.

# List all users
GET  /api/v1/users                  → paginated list with last_login_at

# Create user
POST /api/v1/users                  → validates unique email; hashes password; returns 201

# Get / update / delete single user
GET    /api/v1/users/{id}
PUT    /api/v1/users/{id}           → update name, role, is_active
DELETE /api/v1/users/{id}           → hard delete (blocks self-delete)

# Password management
PUT  /api/v1/users/{id}/password    → admin force-set (no current password required)
PUT  /api/v1/users/me               → update own name (any authenticated user)
PUT  /api/v1/users/me/password      → change own password (requires current password)

Self-protection at the backend level

@router.delete("/users/{user_id}")
async def delete_user(
    user_id: uuid.UUID,
    current_user: User = Depends(require_admin),
    db: AsyncSession = Depends(get_db),
):
    if user_id == current_user.id:
        raise HTTPException(400, "Cannot delete your own account")
    # Proceed with deletion
    ...

The same guard applies to role changes and deactivation — backend-level protection is independent of the UI guards and cannot be bypassed via direct API calls.

3. Documentation system

Backend

File: backend/app/api/v1/endpoints/docs_content.py

Serves markdown files from /app/docs/ (mounted read-only from ./docs on the host). Path traversal is blocked by resolving the target path and verifying it starts with the docs directory.

DOCS_DIR = Path("/app/docs")

@router.get("/manifest")
async def get_manifest(_: UserId) -> dict:
    manifest_path = DOCS_DIR / "manifest.json"
    if manifest_path.exists():
        return json.loads(manifest_path.read_text())
    return _DEFAULT_MANIFEST    # fallback if manifest.json is missing

@router.get("/file")
async def get_file(_: UserId, path: str = Query(...)) -> dict:
    target = (DOCS_DIR / path).resolve()
    # Block path traversal (e.g. ../../etc/passwd)
    if not str(target).startswith(str(DOCS_DIR.resolve())):
        raise HTTPException(403, "Access denied")
    if not target.exists():
        raise HTTPException(404, "File not found")
    return {"path": path, "content": target.read_text(encoding="utf-8")}

Export logic

// Single page MD download
function exportCurrentMd(content: string, filename: string) {
  const blob = new Blob([content], { type: 'text/markdown' })
  const url = URL.createObjectURL(blob)
  const a = document.createElement('a')
  a.href = url
  a.download = `${filename}.md`
  a.click()
  URL.revokeObjectURL(url)
}

// All pages combined MD download
async function exportAllMd(manifest: ManifestNode[], fetchFn: Function) {
  const leaves = collectLeaves(manifest)    // flatten tree to leaf nodes
  const sections = await Promise.all(
    leaves.map(leaf => fetchFn(leaf.path).then(r => ({ label: leaf.label, content: r.content })))
  )
  const combined = sections
    .map(s => `# ${s.label}\n\n${s.content}`)
    .join('\n\n---\n\n')
  downloadBlob(combined, 'InvestPro-Documentation.md', 'text/markdown')
}
