2. Excel service (portfolio_excel.py)

2.1 Path resolution

Two paths are used: the source path (read-only original Excel on the host volume) and the working path (writable copy inside the container). The Settings → App Configuration UI writes to /app/uploads/app_config.json which takes priority over environment variables.

def _source_path() -> Path:
    cfg = get_app_config()   # reads /app/uploads/app_config.json
    return Path(cfg.get("excel_source_path") or settings.investment_excel_source_path)

def _working_path() -> Path:
    cfg = get_app_config()
    return Path(cfg.get("excel_working_path") or settings.investment_excel_path)

2.2 Cache system

Each Uvicorn worker maintains an in-memory _CACHE dict. Entries expire when the .cache_bust sentinel file's mtime is newer than the entry's timestamp. This provides cross-worker cache invalidation without a shared cache layer.

_CACHE: dict = {}
_CACHE_BUST_FILE = Path("/app/uploads/.cache_bust")

def _cache_bust_ts() -> float:
    try:
        return _CACHE_BUST_FILE.stat().st_mtime
    except OSError:
        return 0.0

def _write_cache_bust():
    _CACHE_BUST_FILE.parent.mkdir(parents=True, exist_ok=True)
    _CACHE_BUST_FILE.write_text(str(time.time()))

def _cached(key: str, ttl: float, fn: Callable) -> Any:
    now = time.time()
    bust = _cache_bust_ts()
    entry = _CACHE.get(key)
    if entry and (now - entry[0] < ttl) and (entry[0] > bust):
        return entry[1]    # cache hit
    val = fn()             # recompute
    _CACHE[key] = (now, val)
    return val

Cache keys include the function name and any relevant parameters. TTL for the main DataFrame is 300 seconds; live price fetches use a shorter TTL.

2.3 DataFrame loading

def _load_df() -> pd.DataFrame:
    path = _ensure_working_copy()   # copies from source if working copy doesn't exist
    df = pd.read_excel(str(path), sheet_name="Sheet1")

    # Normalise column names (strip whitespace)
    df.columns = df.columns.str.strip()

    # Parse date columns
    for col in ["Entry Date", "Exit Date"]:
        df[col] = pd.to_datetime(df[col], errors="coerce")

    # Detect open positions: Exit Price is NaN or the string "NOT SELL"
    df["_is_open"] = (
        df["Exit Price"].isna() |
        (df["Exit Price"].astype(str).str.strip().str.upper() == "NOT SELL")
    )
    return df

2.4 Live prices

Live prices are fetched from Yahoo Finance's quote API directly (not via the yfinance library) for lower latency. Thai SET stocks get the .BK suffix.

def fetch_live_prices(symbols: list[str]) -> dict[str, float]:
    # Append .BK for Thai SET stocks
    tickers = [s + ".BK" if not s.endswith(".BK") else s for s in symbols]
    quotes = _yahoo_quote_direct(tickers)
    return {
        s: quotes.get(s + ".BK", {}).get("regularMarketPrice")
        for s in symbols
    }

2.5 P&L calculation

P&L is calculated per row based on direction (Long or Short). Open positions use the live price; closed positions use the Excel Exit Price.

def _calc_pnl(row: pd.Series, live_prices: dict[str, float]) -> tuple[float, float, float]:
    symbol = row["Symbol"]
    entry  = float(row["Entry Price"])
    size   = int(row["Position Size"])
    is_long = "short" not in str(row.get("Direction", "")).lower()

    if row["_is_open"]:
        current = live_prices.get(symbol, entry)   # fallback to entry if no live price
    else:
        current = float(row["Exit Price"])

    if is_long:
        net_pnl = (current - entry) * size
    else:
        net_pnl = (entry - current) * size   # short: profit when price falls

    pnl_pct = (net_pnl / (entry * size)) * 100 if entry and size else 0
    return current, net_pnl, pnl_pct

2.6 Performance aggregation

Daily performance groups positions by their "bucket date" (Exit Date for closed, today for open) and computes cumulative P&L over the date range.

def get_daily_performance(from_date, to_date, period="daily") -> list[dict]:
    date_range = pd.date_range(
        from_date or df["_bucket_date"].min(),
        to_date or date.today(),
        freq="D"
    )
    grouped = df.groupby("_bucket_date")["_net_pnl"].sum()
    result = []
    cumulative = 0.0
    for d in date_range:
        daily = float(grouped.get(d.date(), 0.0))
        cumulative += daily
        result.append({
            "date": str(d.date()),
            "label": _fmt_period_label(d, period),
            "dailyPnl": daily,
            "cumulativePnl": cumulative
        })
    return result

3. API endpoints

All endpoints use the prefix /api/v1/portfolio-tracker and require a valid Bearer token via the UserId dependency.

GET /performance/by-dateReturns [{period, label, net, accumulatedPnl, wins, losses, total, winRate}]

GET /market/global-indicesSame for global tickers

Refresh modal pattern

The refresh modal mounts via a boolean flag. It runs all async work inside a useEffect on mount rather than in an event handler. This is necessary because React 18 batches state updates inside async functions — running async work inline in a click handler would prevent intermediate state updates (like progress steps) from rendering.

// Parent component
const [showRefresh, setShowRefresh] = useState(false)
{showRefresh && <RefreshModal onClose={() => setShowRefresh(false)} />}

// RefreshModal — runs async work after mount
useEffect(() => {
  let cancelled = false
  const run = async () => {
    setPhase('copying')
    try {
      const res = await portfolioTrackerService.refresh()
      if (cancelled) return
      setPhase('reloading')
      await Promise.all([
        queryClient.refetchQueries({ queryKey: ['portfolio-positions'] }),
        queryClient.refetchQueries({ queryKey: ['portfolio-performance'] }),
      ])
      if (!cancelled) setPhase('done')
    } catch (e) {
      if (!cancelled) setPhase('error')
    }
  }
  run()
  return () => { cancelled = true }   // cleanup on unmount
}, [])
